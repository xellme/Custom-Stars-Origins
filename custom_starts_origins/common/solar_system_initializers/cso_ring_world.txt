				count = 2

					type = "medium_debris_01_object"

        orbit_distance = 50
		deposit_blockers = none
        modifiers = none

		init_effect = {
            generate_empire_home_planet = yes		    
		}
	}

	init_effect = {
		every_system_planet = {
			clear_deposits = yes
        }

			type = ring_world_ruined
