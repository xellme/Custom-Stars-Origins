# Habitat trigger
country_event = {
	id = cso_habitat.10
	hide_window = yes
    is_triggered_only = yes

    trigger = {
        exists = capital_scope
        capital_scope.solar_system = {
            OR = {
                has_star_flag = cso_empire_home_system_habitat
                has_star_flag = cso_empire_home_system_habitat_tech
            }
        }
    }

	immediate = {
        country_event = {
            id = cso_habitat.11
        }
        country_event = {
            id = cso_habitat.12 
        }
	}
}

# change planet to habitat
planet_event = {
	id = cso_habitat.101
	hide_window = yes
    is_triggered_only = yes

	immediate = {
        # clear generic planet stuff
        clear_deposits = yes
        clear_blockers = yes
        remove_all_districts = yes

        # clear specyfic planet stuff
        remove_building = building_research_lab_1
        remove_building = building_commercial_zone

        # planet type
        change_pc = pc_habitat
        set_planet_flag = megastructure
        set_planet_flag = habitat

        # districts
        add_district = district_hab_housing
        add_district = district_hab_housing
        add_district = district_hab_housing
        
        add_district = district_hab_commercial
        add_district = district_hab_commercial
        add_district = district_hab_science

        # food solution
        owner = {
            give_technology = {
                tech = tech_hydroponics
            }
        }

        add_building = building_hydroponics_farm
        add_building = building_hydroponics_farm
    }
}
